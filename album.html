<script>
const sun=document.getElementById("sun");
const panel=document.getElementById("panel");
const fan=document.getElementById("fan");
const raysBox=document.getElementById("rays");
const led=document.getElementById("led");
const batteryLevel=document.getElementById("batteryLevel");
const batteryText=document.getElementById("batteryText");
const rpmMeter=document.getElementById("rpmMeter");

let battery=0;
let solar=false;
let currentRPM=0;

/* RAYS */
const rayCount=16;
const rays=[];
for(let i=0;i<rayCount;i++){
    const r=document.createElement("div");
    r.className="ray";
    raysBox.appendChild(r);
    rays.push(r);
}

/* DRAG */
sun.onpointerdown=e=>{
    sun.setPointerCapture(e.pointerId);
    sun.dx=e.clientX-sun.offsetLeft;
    sun.dy=e.clientY-sun.offsetTop;
};
sun.onpointermove=e=>{
    if(!sun.hasPointerCapture(e.pointerId))return;
    sun.style.left=e.clientX-sun.dx+"px";
    sun.style.top=e.clientY-sun.dy+"px";
};
sun.onpointerup=e=>sun.releasePointerCapture(e.pointerId);

/* RAY vs RECT */
function rayHitsRect(x,y,dx,dy,r){
    let t1=(r.left-x)/dx,t2=(r.right-x)/dx;
    let t3=(r.top-y)/dy,t4=(r.bottom-y)/dy;
    let tmin=Math.max(Math.min(t1,t2),Math.min(t3,t4));
    let tmax=Math.min(Math.max(t1,t2),Math.max(t3,t4));
    return tmax>=0&&tmin<=tmax;
}

/* MAIN LOOP */
setInterval(()=>{
    const s=sun.getBoundingClientRect();
    const p=panel.getBoundingClientRect();
    const cx=s.left+s.width/2;
    const cy=s.top+s.height/2;

    solar=false;
    rays.forEach((ray,i)=>{
        const a=i/rayCount*Math.PI*2;
        const dx=Math.cos(a),dy=Math.sin(a);
        ray.style.width="1000px";
        ray.style.left=cx+"px";
        ray.style.top=cy+"px";
        ray.style.transform=`rotate(${a}rad)`;
        if(rayHitsRect(cx,cy,dx,dy,p)) solar=true;
    });

    if(solar){
        battery=Math.min(100,battery+0.3); // unchanged
        led.classList.add("on");
    }else{
        led.classList.remove("on");
    }

    if(currentRPM>0 && battery>0){
        battery-=currentRPM/100000;        // unchanged
    }

    if(battery<=0){
        battery=0;
        stopFan();
    }

    batteryLevel.style.width=battery+"%";
    batteryText.innerText="Battery: "+battery.toFixed(0)+"%";
},100);

/* FAN ‚Äî ONLY SPEED √ó10 */
function setRPM(rpm){
    if(battery<=0)return;
    currentRPM=rpm;

    /* üî• ONLY CHANGE IS HERE */
    const normalDuration = Math.max(0.3, 6 - Math.log10(rpm));
    fan.style.animationDuration = (normalDuration / 10) + "s";

    fan.style.animationPlayState="running";
    rpmMeter.innerText="RPM: "+rpm;
}
function stopFan(){
    currentRPM=0;
    fan.style.animationPlayState="paused";
    rpmMeter.innerText="RPM: 0";
}
</script>
